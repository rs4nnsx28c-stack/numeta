  <input type="number" id="meses" placeholder="Meses">

  <button onclick="calcular()">Calcular</button>

  <div class="resultado" id="resultado"></div>
</section>

<section>
  <h2>Progresso</h2>

  <input type="number" id="guardado" placeholder="Já guardado (R$)">

  <button onclick="progresso()">Ver progresso</button>

  <div class="resultado" id="progressoResultado"></div>
</section>

<script>
  function calcular() {
    let total = parseFloat(document.getElementById("valorTotal").value);
    let meses = parseFloat(document.getElementById("meses").value);

    if (total > 0 && meses > 0) {
      let porMes = total / meses;
      document.getElementById("resultado").innerText =
        "Guardar por mês: R$ " + porMes.toFixed(2);
    } else {
      document.getElementById("resultado").innerText =
        "Preencha corretamente.";
    }
  }

  function progresso() {
    let total = parseFloat(document.getElementById("valorTotal").value);
    let guardado = parseFloat(document.getElementById("guardado").value);

    if (total > 0 && guardado >= 0) {
      let porcentagem = (guardado / total) * 100;
      document.getElementById("progressoResultado").innerText =
        "Progresso: " + porcentagem.toFixed(1) + "%";
    } else {
      document.getElementById("progressoResultado").innerText =
        "Preencha corretamente.";
    }
  }
</script>
